📋 首次使用必看：點此展開 Supabase 建表 SQL

複製以下 SQL，到 Supabase → SQL Editor → New query 貼上並執行：

-- ① 項目資料表
create table projects (
  id text primary key,
  label text not null,
  color_idx integer not null default 0,
  sort_order integer not null default 0,
  initial_capital numeric not null default 0,
  current_balance numeric not null default 0,
  slider_pct numeric not null default 5,
  manual_bet numeric not null default 0,
  manual_pct numeric not null default 0
);

-- ② 下注紀錄資料表
create table betting_records (
  id bigint primary key,
  project_id text,
  date text not null,
  project text not null,
  init_bal numeric not null default 0,
  pct numeric not null default 0,
  bet numeric not null default 0,
  pnl numeric,
  after_bal numeric,
  status text not null default 'pending',
  note text default '',
  odds numeric not null default 1.0
);

-- ※ 若資料表已存在，請執行以下語句補加 odds 欄位：
-- alter table betting_records add column if not exists odds numeric not null default 1.0;

-- ③ 應用程式設定表（儲存備註標籤等）
create table app_settings (
  key text primary key,
  value jsonb not null
);

-- ④ 開放 anon key 讀寫（個人工具）
alter table projects enable row level security;
alter table betting_records enable row level security;
alter table app_settings enable row level security;
create policy "anon_all" on projects
  for all using (true) with check (true);
create policy "anon_all" on betting_records
  for all using (true) with check (true);
create policy "anon_all" on app_settings
  for all using (true) with check (true);
